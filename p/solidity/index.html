    function verifySignature(address userAddress, bytes memory signature) public pure returns (bool) {
        bytes32 messageHash = keccak256(abi.encodePacked(userAddress));
        address recoveredAddress = ecrecover(messageHash, signature);
        return recoveredAddress == userAddress;
    }
}
